## What Is RAG and Why Does It Matter for Security?

Retrieval-Augmented Generation (RAG) grounds LLM responses in your own data rather than relying solely on the model's training knowledge. This is critical for enterprise AI because:

1. **Accuracy**: LLM training data is outdated; your data is current
2. **Control**: You decide what knowledge the AI can access
3. **Auditability**: You can trace which documents informed a response
4. **Security**: You limit the AI's knowledge to authorized content

But RAG also introduces new attack surfaces — vector poisoning, retrieval manipulation, and grounding bypass attacks.

---

## AWS Bedrock Knowledge Bases Architecture

```
[Data Source] → [S3 Bucket] → [Bedrock KB] → [Embedding Model] → [Vector Store]
                                    ↑
                               [Ingestion Job]
                                    ↓
[User Query] → [Bedrock Agent] → [KB Retrieve & Generate] → [Response]
                    ↑
               [Guardrails]
```

### Supported Data Sources

- **Amazon S3** (PDF, Word, TXT, HTML, Markdown, CSV, JSON)
- **Confluence** (pages and spaces)
- **Salesforce** (articles, cases)
- **SharePoint** (sites and libraries)
- **Web crawler** (public URLs)

## Setting Up a Knowledge Base

```python
import boto3
import json

bedrock_agent = boto3.client("bedrock-agent", region_name="us-east-1")

def create_security_knowledge_base() -> dict:
    """Create a Knowledge Base for security documentation."""
    
    response = bedrock_agent.create_knowledge_base(
        name="acme-security-kb",
        description="Security policies, runbooks, compliance documentation, and threat intelligence",
        roleArn="arn:aws:iam::123456789012:role/BedrockKBServiceRole",
        
        knowledgeBaseConfiguration={
            "type": "VECTOR",
            "vectorKnowledgeBaseConfiguration": {
                "embeddingModelArn": "arn:aws:bedrock:us-east-1::foundation-model/amazon.titan-embed-text-v2:0",
                "embeddingModelConfiguration": {
                    "bedrockEmbeddingModelConfiguration": {
                        "dimensions": 1024,
                        "embeddingDataType": "FLOAT32",
                    }
                },
                "supplementalDataStorageConfiguration": {
                    "supplementalDataStorageLocations": [
                        {
                            "type": "S3",
                            "s3Location": {
                                "uri": "s3://acme-kb-storage/supplemental/"
                            }
                        }
                    ]
                }
            }
        },
        
        storageConfiguration={
            "type": "OPENSEARCH_SERVERLESS",
            "opensearchServerlessConfiguration": {
                "collectionArn": "arn:aws:aoss:us-east-1:123456789012:collection/security-kb-001",
                "vectorIndexName": "security-docs",
                "fieldMapping": {
                    "vectorField": "embedding",
                    "textField": "content",
                    "metadataField": "metadata"
                }
            }
        },
        
        tags={"Environment": "Production", "Team": "Security", "CostCenter": "SECURITY-001"}
    )
    
    return response["knowledgeBase"]


def add_s3_data_source(kb_id: str, s3_bucket: str) -> dict:
    """Add an S3 data source to the Knowledge Base."""
    
    return bedrock_agent.create_data_source(
        knowledgeBaseId=kb_id,
        name="security-docs-s3",
        description="Security documentation from S3",
        
        dataSourceConfiguration={
            "type": "S3",
            "s3Configuration": {
                "bucketArn": f"arn:aws:s3:::{s3_bucket}",
                "inclusionPrefixes": ["security-policies/", "runbooks/", "compliance/"],
                # Explicitly exclude sensitive raw data
                "bucketOwnerAccountId": "123456789012",
            }
        },
        
        vectorIngestionConfiguration={
            "chunkingConfiguration": {
                "chunkingStrategy": "SEMANTIC",
                "semanticChunkingConfiguration": {
                    "maxTokens": 300,
                    "bufferSize": 1,
                    "breakpointPercentileThreshold": 95,
                }
            },
            "parsingConfiguration": {
                "parsingStrategy": "BEDROCK_FOUNDATION_MODEL",
                "bedrockFoundationModelConfiguration": {
                    "modelArn": "arn:aws:bedrock:us-east-1::foundation-model/anthropic.claude-3-haiku-20240307-v1:0",
                    "parsingPrompt": {
                        "parsingPromptText": "Extract all text content from this document. For tables, preserve column headers. For images, describe security-relevant content."
                    }
                }
            }
        },
        
        dataDeletionPolicy="DELETE"  # Clean up vectors when document is deleted
    )
```

## Secure Retrieval Pattern

```python
from typing import Optional

class SecureKBRetriever:
    def __init__(self, kb_id: str, guardrail_id: str, guardrail_version: str):
        self.kb_id = kb_id
        self.guardrail_id = guardrail_id
        self.guardrail_version = guardrail_version
        self.runtime = boto3.client("bedrock-agent-runtime", region_name="us-east-1")
    
    def retrieve_and_generate(
        self,
        query: str,
        user_id: str,
        session_id: str,
        max_results: int = 5,
        filter_tags: Optional[list[str]] = None,
    ) -> dict:
        """Retrieve from KB and generate a grounded response."""
        
        # Build retrieval filter for access control
        retrieval_filter = self._build_access_filter(user_id, filter_tags)
        
        response = self.runtime.retrieve_and_generate(
            input={"text": query},
            retrieveAndGenerateConfiguration={
                "type": "KNOWLEDGE_BASE",
                "knowledgeBaseConfiguration": {
                    "knowledgeBaseId": self.kb_id,
                    "modelArn": "arn:aws:bedrock:us-east-1::foundation-model/anthropic.claude-3-5-sonnet-20241022-v2:0",
                    "retrievalConfiguration": {
                        "vectorSearchConfiguration": {
                            "numberOfResults": max_results,
                            "filter": retrieval_filter,
                            "overrideSearchType": "HYBRID",  # Semantic + keyword
                        }
                    },
                    "generationConfiguration": {
                        "promptTemplate": {
                            "textPromptTemplate": """
You are a security documentation assistant. Answer questions using ONLY the provided context.

$search_results$

Question: $query$

Instructions:
- Base your response ONLY on the provided context
- If the context does not contain enough information, say so explicitly
- Cite the source document for each claim
- If you see conflicting information across sources, note the discrepancy
"""
                        },
                        "guardrailConfiguration": {
                            "guardrailId": self.guardrail_id,
                            "guardrailVersion": self.guardrail_version,
                        }
                    },
                }
            },
            sessionId=session_id,
        )
        
        return {
            "answer": response["output"]["text"],
            "citations": self._extract_citations(response),
            "session_id": session_id,
        }
    
    def _build_access_filter(
        self,
        user_id: str,
        allowed_tags: Optional[list[str]]
    ) -> dict:
        """Build a metadata filter for user-level access control."""
        
        # Get user's clearance level from your auth system
        user_clearance = self._get_user_clearance(user_id)
        
        conditions = [
            {
                "equals": {
                    "key": "clearance_level",
                    "value": user_clearance
                }
            }
        ]
        
        if allowed_tags:
            conditions.append({
                "in": {
                    "key": "document_type",
                    "value": allowed_tags
                }
            })
        
        return {"andAll": conditions} if len(conditions) > 1 else conditions[0]
    
    def _get_user_clearance(self, user_id: str) -> str:
        # Implement your authorization logic
        return "PUBLIC"  # Default to most restrictive
    
    def _extract_citations(self, response: dict) -> list[dict]:
        citations = []
        for citation in response.get("citations", []):
            for ref in citation.get("retrievedReferences", []):
                citations.append({
                    "content": ref.get("content", {}).get("text", "")[:200],
                    "source": ref.get("location", {}).get("s3Location", {}).get("uri", ""),
                    "score": ref.get("score"),
                })
        return citations
```

## Summary

AWS Bedrock Knowledge Bases provide a managed RAG infrastructure that handles embedding, vector storage, and retrieval. The critical security considerations: **access control at the retrieval layer** (metadata filters), **grounding guardrails** to prevent hallucination injection, **document-level access policies** (not just collection-level), and **audit logging** of all retrieval operations. RAG systems are only as secure as the weakest link in the chain — document ingestion, vector storage, retrieval, and generation must each be secured independently.
